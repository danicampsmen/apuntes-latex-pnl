\section*{Ejercicios del Capítulo 3}

\begin{exercise}\label{v2:ejer:3.1.1}
    Probar los siguientes hechos, que pueden ser pensados como caracterizaciones adicionales de direcciones de descenso. Si la función $f : \Rn \to \R$ es dos veces diferenciable en el punto $x \in \Rn$, entonces
    \begin{enumerate}[(a)]
        \item Para todo $d \in \mathcal{D}_f(x)$ tal que $\interno{f'(x)}{d} = 0$, se tiene que $\interno{f''(x)d}{d} \le 0$.
        \item Si $d \in \Rn$ satisface $\interno{f'(x)}{d} = 0$ e $\interno{f''(x)d}{d} < 0$, se tiene que $d \in \mathcal{D}_f(x)$.
    \end{enumerate}
\end{exercise}

\begin{exercise}\label{v2:ejer:3.1.2}
    Suponiendo que $f : \Rn \to \R$ sea continuamente diferenciable en el punto $x^k$ y que $d^k \in \mathcal{D}_f(x^k)$, probar que la modificación de la regla de Armijo donde la desigualdad \eqref{v2:eq:3.6} es sustituida por
    \[
        f(x^k + \alpha d^k) \le f(x^k) + \sigma \alpha^2 \interno{f'(x^k)}{d^k},
    \]
    con $\sigma > 0$, está bien definida (i.e., que existe $\bar{\alpha}_k > 0$ tal que la condición de arriba vale para todo $\alpha \in (0, \bar{\alpha}_k]$).
\end{exercise}

\begin{exercise}\label{v2:ejer:3.1.6}
    Considerar el esquema iterativo dado por
    \[
        x^{k+1} = x^k + \alpha_k d^k, \quad k = 0, 1, \dots,
    \]
    donde
    \[
        c_1 \|f'(x^k)\|^{p_1} \le -\interno{f'(x^k)}{d^k}, \quad \|d^k\| \le c_2 \|f'(x^k)\|^{p_2},
    \]
    con $c_1, c_2 > 0$, $p_1, p_2 \ge 0$, y la longitud de paso $\alpha_k$ es calculada por la regla de minimización unidimensional o por la regla de Armijo.
    
    Suponiendo que la función $f$ sea continuamente diferenciable, probar que todo punto de acumulación de la secuencia $\{x^k\}$ es un punto estacionario del problema \eqref{v2:eq:3.1}.
\end{exercise}

\begin{exercise}\label{v2:ejer:3.1.9}
    Probar el siguiente resultado, siguiendo el razonamiento de demostración del Lema \ref{v2:lem:3.1.5}.
    
    Supongamos que existan números $\delta > 0$ y $\gamma > 0$ tales que la función $f : \Rn \to \R$ sea tres veces diferenciable en $U = U(S, \delta)$, con la derivada tercera acotada en $U$. Supongamos que \eqref{v2:eq:3.42} sea satisfecha. Entonces para todo $\nu \in (0, 4)$, existe $r > 0$ tal que
    \[
        \|f'(x)\|^2 \ge \nu \gamma (f(x) - \bar{v}) \quad \forall x \in U(S, r).
    \]
\end{exercise}

\begin{exercise}\label{v2:ejer:3.1.14}
    Sea $\{x^k\}$ una secuencia generada por
    \[
        x^{k+1} = x^k + \alpha_k d^k, \quad k = 0, 1, \dots,
    \]
    donde
    \[
        c_1 \|f'(x^k)\|^2 \le -\interno{f'(x^k)}{d^k}, \quad \|d^k\| \le c_2 \|f'(x^k)\|,
    \]
    $c_1, c_2 > 0$, y la secuencia $\{\alpha_k\} \subset \R_+$ satisface
    \[
        \lim_{k \to \infty} \alpha_k = 0, \quad \sum_{k=0}^\infty \alpha_k = +\infty.
    \]
    Sea $f$ acotada inferiormente y sea la derivada de $f$ Lipschitz-continua en $\Rn$. Probar que la secuencia $\{f(x^k)\}$ converge y que
    \[
        \lim \inf_{k \to \infty} \|f'(x^k)\| = 0.
    \]
    Más aún, si conjuntos de nivel $\{x \in \Rn \mid f(x) \le c\}$ son acotados para todo $c \in \R$, entonces la secuencia $\{x^k\}$ es acotada y todo punto de acumulación de $\{x^k\}$ es punto estacionario del problema \eqref{v2:eq:3.1}.
\end{exercise}

\begin{exercise}\label{v2:ejer:3.1.15}
    Sea $\{x^k\}$ una secuencia generada por
    \[
        x^{k+1} = x^k + \alpha_k d^k, \quad k = 0, 1, \dots,
    \]
    donde $\alpha_k$ es calculado por la regla de Armijo,
    \[
        d^k = -(0, \dots, 0, f'_{x_i}(x^k), 0, \dots, 0),
    \]
    y el índice $i \in \{1, \dots, n\}$ satisface
    \[
        |f'_{x_i}(x^k)| = \max_{j=1, \dots, n} |f'_{x_j}(x^k)|.
    \]
    Suponiendo que la función $f$ es continuamente diferenciable, probar que todo punto de acumulación de $\{x^k\}$ es punto estacionario del problema \eqref{v2:eq:3.1}.
\end{exercise}

\begin{exercise}\label{v2:ejer:3.1.16}

        \lim \inf_{i \to \infty} \interno{f'(x^{k_i})}{d^{k_i}} < 0.
    \]
    Sea $\{x^k\}$ una secuencia generada por

        x^{k+1} = x^k + \alpha_k d^k, \quad k = 0, 1, \dots,

    donde $\alpha_k$ es calculado por la regla de Armijo. Probar que todo punto de acumulación de $\{x^k\}$ es punto estacionario del problema \eqref{v2:eq:3.1}.
\end{exercise}

\begin{exercise}\label{v2:ejer:3.1.17}

\begin{exercise}\label{v2:ejer:3.2.5}
    Probar la siguiente afirmación.
    
    Sea la función $\Phi : \R \to \R$ diferenciable $p$ veces en el punto $\bar{x} \in \R$, $p \ge 2$, donde $\bar{x}$ es una raíz de orden $p$ de la ecuación \eqref{v2:eq:3.49}, i.e.,
    \[
        \Phi(\bar{x}) = \Phi'(\bar{x}) = \dots = \Phi^{(p-1)}(\bar{x}) = 0, \quad \Phi^{(p)}(\bar{x}) \neq 0.
    \]
    Entonces el método de Newton converge localmente a $\bar{x}$ con tasa lineal, y si introducimos una longitud de paso igual a $p$, la tasa de convergencia se vuelve superlineal.
\end{exercise}

\begin{exercise}\label{v2:ejer:3.2.6}
    Probar que el método de Newton es invariante en relación a transformaciones lineales de variables, en el sentido siguiente.
    
    Sea $x = Ay$ una transformación de variables, dada por una matriz no singular $A \in \R^{n \times n}$. Entonces si el método de Newton en variables $y$ genera una secuencia $\{y^k\}$ y en variables $x$ genera una secuencia $\{x^k\}$, se tiene que $y^k = A^{-1}x^k$.
\end{exercise}

\begin{exercise}\label{v2:ejer:3.2.12}
    Mostrar que los métodos DFP y BFGS tienen una relación de ``dualidad'' en el sentido siguiente.
    
    Para una matriz simétrica definida positiva $Q_k \in \R^{n \times n}$, definimos $H_k = Q_k^{-1}$. Sea la matriz $Q_{k+1}$ generada por \eqref{v2:eq:3.76}, y la matriz $H_{k+1}$ generada por
    \[
        H_{k+1} = H_k + \frac{s^k (s^k)^\top}{\interno{r^k}{s^k}} - \frac{(H_k r^k)(H_k r^k)^\top}{\interno{H_k r^k}{r^k}}
    \]
    (comparar con \eqref{v2:eq:3.72}), suponiendo que $H_{k+1}$ sea no singular. Entonces la matriz $Q_{k+1}$ es no singular y se tiene que $H_{k+1} = Q_{k+1}^{-1}$.
    
    Utilizar este hecho para probar, para BFGS, el resultado análogo a aquel de la Proposición \ref{v2:prop:3.2.1}.
\end{exercise}

\begin{exercise}\label{v2:ejer:3.3.1}
    Sea $f : \Rn \to \R$ una función dos veces continuamente diferenciable. Considerar el siguiente algoritmo.
    
    Elegir $x^0 \in \Rn$ y tomar $k := 0$. Elegir los parámetros $\sigma \in (0, 1/2)$, $\theta \in (0, 1)$. Si $f''(x^k)$ es no singular, tomar $d^k = -(f''(x^k))^{-1} f'(x^k)$. Caso contrario, tomar $d^k = 0$. Tomar $h^k = -f'(x^k)$. Computar
    \[
        x^{k+1} = x^k + \alpha_k ((1 - \alpha_k)h^k + \alpha_k d^k),
    \]
    donde $\alpha_k$ es el mayor entre todos los números de la forma $\theta^s$, $s \in \N$, satisfaciendo
    \[
        f(x^k + \theta^s ((1 - \theta^s)h^k + \theta^s d^k)) \le f(x^k) - \sigma \theta^s \interno{f'(x^k)}{(1 - \theta^s)h^k + \theta^s d^k}.
    \]
    Probar que el método está bien definido (en particular, la regla del cálculo de la longitud del paso está bien definida). Probar que todo punto de acumulación de una secuencia $\{x^k\}$ así generada, es un punto estacionario del problema \eqref{v2:eq:3.77}. Probar que si $\{x^k\}$ converge a un punto estacionario $\bar{x}$ donde $f''(\bar{x})$ es definida positiva, entonces la tasa de convergencia de $\{x^k\}$ a $\bar{x}$ es superlineal.
\end{exercise}

\begin{exercise}\label{v2:ejer:3.3.2}
    Sean dados un vector $q \in \Rn$, una matriz simétrica $Q \in \R^{n \times n}$, y un número $\delta > 0$. Probar que si $\tilde{x} \in \Rn$ es una solución global del problema
    \begin{equation}
        \min \interno{Qx}{x} + \interno{q}{x} \quad \text{sujeto a } x \in B(0, \delta),
        \label{v2:eq:3.87}
    \end{equation}
    entonces existe un único número $\mu \ge 0$ tal que
    \[
        2(Q + \mu E^n)\tilde{x} + q = 0, \quad \mu(\|\tilde{x}\| - \delta) = 0,
    \]
    donde la matriz $Q + \mu E^n$ es semidefinida positiva.
    
    Si la matriz $Q$ es definida positiva, entonces
    \[
        \tilde{x} = \begin{cases}
            -Q^{-1}q / 2, & \text{si } \|Q^{-1}q\|/2 \le \delta, \\
            -(Q + \mu E^n)^{-1}q / 2, & \text{si no,}
        \end{cases}
    \]
    donde $\mu \ge 0$ está unívocamente definido por la ecuación
    \[
        \|(Q + \mu E^n)^{-1}q\| / 2 = \delta.
    \]
\end{exercise}

\begin{exercise}\label{v2:ejer:3.3.3}
    En la notación del Ejercicio \ref{v2:ejer:3.3.2}, sean $\tilde{x} \in \Rn$ y $\mu \in \R$ tales que $2(Q + \mu E^n)\tilde{x} + q = 0$ y la matriz $Q + \mu E^n$ es semidefinida positiva. Probar las siguientes afirmaciones:
    \begin{enumerate}[(a)]
        \item Si $\|\tilde{x}\| \le \delta$ y $\mu = 0$, entonces $\tilde{x}$ es una solución global del problema \eqref{v2:eq:3.87}.
        \item Si $\|\tilde{x}\| = \delta$, entonces $\tilde{x}$ es una solución global del problema
        \[
            \min \interno{Qx}{x} + \interno{q}{x} \quad \text{sujeto a } x \in \{x \in \Rn \mid \|x\| = \delta\}.
        \]
        \item Si $\|\tilde{x}\| = \delta$ y $\mu \ge 0$, entonces $\tilde{x}$ es una solución global del problema \eqref{v2:eq:3.87}.
    \end{enumerate}
    Además de esto, si la matriz $Q + \mu E^n$ es definida positiva, en todas las afirmaciones (a)-(c) el punto $\tilde{x}$ es la única solución global del problema correspondiente.
\end{exercise}

\begin{exercise}\label{v2:ejer:3.3.4}

\begin{exercise}\label{v2:ejer:3.4.4}
