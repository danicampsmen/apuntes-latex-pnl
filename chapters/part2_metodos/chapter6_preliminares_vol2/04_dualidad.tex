\section{Dualidad}
% =========================================================

Recordamos ahora algunos hechos de la teoría de dualidad. Consideramos un problema en formato general
\begin{equation}
    \min f(x) \quad \text{sujeto a} \quad x \in D = \left\{ x \in \Omega \mathrel{\bigg|} \begin{array}{l} h(x) = 0, \\ g(x) \le 0 \end{array} \right\},
    \label{v2:eq:1.34}
\end{equation}
Llamamos a \eqref{v2:eq:1.34} el \textbf{problema primal}. La Lagrangiana de este problema está dada por $L(x, \lambda, \mu) = f(x) + \langle \lambda, h(x) \rangle + \langle \mu, g(x) \rangle$.

El \textbf{problema dual} se obtiene intercambiando el orden de la minimización y de la maximización, lo que resulta en:
\begin{equation}
    \max \varphi(\lambda, \mu) \quad \text{sujeto a} \quad (\lambda, \mu) \in \Delta,
    \label{v2:eq:1.37}
\end{equation}
donde $\varphi(\lambda, \mu) = \inf_{x \in \Omega} L(x, \lambda, \mu)$ es la función dual y $\Delta$ es su dominio de valores finitos.

\begin{theorem}\label{v2:thm:1.4.1}
    Para cualquier problema primal del tipo \eqref{v2:eq:1.34}, el conjunto factible $\Delta$ del problema dual \eqref{v2:eq:1.37} es convexo y la función dual $\varphi : \Delta \to \R$ es cóncava.
\end{theorem}

\begin{proof}
    Véase \cite{izmailov2020otimizacaovolume1} (o \cref{v1:thm:propiedades_dual}).
\end{proof}

\begin{theorem}[El Teorema de la dualidad débil]\label{v2:thm:1.4.2}
    Para cualquier par de problemas primal y dual, se tiene que $\varphi(\lambda, \mu) \le f(x)$ para todo $x \in D$ y todo $(\lambda, \mu) \in \Delta$.
\end{theorem}

\begin{proof}
    Véase \cite{izmailov2020otimizacaovolume1} (o \cref{v1:thm:dualidad_debil_general}).
\end{proof}

\begin{theorem}[Teorema de la dualidad fuerte]\label{v2:thm:1.4.3}
    Sean $\Omega = \R^n$, $f : \R^n \to \R$ y $g : \R^n \to \R^m$ funciones convexas y $h : \R^n \to \R^l$ una función afín. Supongamos que el problema primal satisface la condición de Slater.
    Si el valor óptimo de \eqref{v2:eq:1.34} es finito, entonces el problema dual posee una solución y no hay brecha de dualidad ($\bar{v} = \bar{w}$).
\end{theorem}

\begin{proof}
    Véase \cite{izmailov2020otimizacaovolume1} (o \cref{v1:thm:dualidad_fuerte_slater}).
\end{proof}

\begin{theorem}[Teorema de Kuhn-Tucker]\label{v2:thm:1.4.4}
    Bajo las hipótesis del Teorema \ref{v2:thm:1.4.3}, $\bar{x} \in D$ es una solución del problema primal \eqref{v2:eq:1.34} si, y solamente si, existe $(\bar{\lambda}, \bar{\mu}) \in \R^l \times \R^m_+$ tal que
    \begin{align}
        L(\bar{x}, \bar{\lambda}, \bar{\mu}) &= \min_{x \in \Omega} L(x, \bar{\lambda}, \bar{\mu}), \label{v2:eq:1.38} \\
        \bar{\mu}_i g_i(\bar{x}) &= 0, \quad i = 1, \dots, m. \label{v2:eq:1.39}
    \end{align}
    El conjunto de los pares $(\bar{\lambda}, \bar{\mu})$ satisfaciendo estas condiciones coincide con las soluciones del problema dual.
\end{theorem}

\begin{proof}
    Véase \cite{izmailov2020otimizacaovolume1} (o \cref{v1:thm:kuhn_tucker_non_diff}).
\end{proof}

\begin{proposition}[Subgradiente de la función dual]\label{v2:prop:1.4.1}
    Sean $(\lambda, \mu) \in \Delta$ y $x(\lambda, \mu) \in \Omega$ tales que
    \[
        L(x(\lambda, \mu), \lambda, \mu) = \varphi(\lambda, \mu) = \min_{x \in \Omega} L(x, \lambda, \mu).
    \]
    Entonces
    \[
        -(h(x(\lambda, \mu)), g(x(\lambda, \mu))) \in \partial(-\varphi)(\lambda, \mu).
    \]
\end{proposition}

\begin{proof}
    Véase \cite{izmailov2020otimizacaovolume1} (o \cref{v1:prop:dual_subgradient}).
\end{proof}
