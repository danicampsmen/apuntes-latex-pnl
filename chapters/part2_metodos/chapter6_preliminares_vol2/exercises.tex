\section*{Ejercicios del Capítulo 6}
% =========================================================

\begin{exercise}\label{v2:ejer:1.5.1}
 Probar el Lema 1.5.3 (Lema de Gronwall, \cref{v2:lem:1.5.3}) utilizando la técnica de inducción.
\end{exercise}

\begin{exercise}\label{v2:ejer:1.6.1}
 Probar que si $F \colon \R^n \to \R^l$ es Lipschitz-continua con módulo $L > 0$ en una vecindad del punto $x \in \R^n$, entonces $\partial F(x)$ es un conjunto compacto no vacío y $\|H\| \le L \quad \forall H \in \partial F(x)$.
\end{exercise}

\begin{exercise}\label{v2:ejer:1.6.2}
 Probar que si la función $F \colon \R^n \to \R$ es convexa en una vecindad (convexa) del punto $x \in \R^n$, entonces $F$ es semi-diferenciable en el punto $x$.
\end{exercise}

\begin{exercise}\label{v2:ejer:1.6.3}
 Probar que si las funciones $F_1, F_2 \colon \R^n \to \R^l$ son semi-diferenciables en el punto $x \in \R^n$, entonces las funciones definidas a continuación también son semi-diferenciables en $x$:
 \begin{enumerate}[(a)]
  \item $F \colon \R^n \to \R$, $F(x) = \langle F_1(x), F_2(x) \rangle$.
  \item $F \colon \R^n \to \R^l$, $F(x) = F_1(x) + F_2(x)$.
  \item $F \colon \R^n \to \R^l$, $F_i(x) = \max\{(F_1(x))_i, (F_2(x))_i\}$, $i = 1, \dots, l$.
  \item $F \colon \R^n \to \R^l$, $F_i(x) = \min\{(F_1(x))_i, (F_2(x))_i\}$, $i = 1, \dots, l$.
 \end{enumerate}
\end{exercise}

\begin{exercise}\label{v2:ejer:1.6.4}
 Probar que si las funciones $F_1 \colon \R^n \to \R^{l_1}$ y $F_2 \colon \R^n \to \R^{l_2}$ son semi-diferenciables en el punto $x$, la función $F \colon \R^n \to \R^{l_1} \times \R^{l_2}$, definida por $F(x) = (F_1(x), F_2(x))$, es semi-diferenciable en $x$.
\end{exercise}

\begin{exercise}\label{v2:ejer:1.6.5}
 Probar que si la función $F_1 \colon \R^n \to \R^{l_1}$ es semi-diferenciable en el punto $x \in \R^n$, y la función $F_2 \colon \R^{l_1} \to \R^{l_2}$ es semi-diferenciable en el punto $F_1(x)$, entonces la función $F \colon \R^n \to \R^{l_2}$, dada por $F(x) = F_2(F_1(x))$, es semi-diferenciable en $x$.
\end{exercise}
