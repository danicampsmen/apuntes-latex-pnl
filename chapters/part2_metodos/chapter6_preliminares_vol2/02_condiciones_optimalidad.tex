\section{Condiciones de optimalidad}
% =========================================================

Cuando $D = \R^n$, decimos que el problema \eqref{v2:eq:1.1} es irrestricto, y cuando $D \neq \R^n$ hablamos de optimización con restricciones.
Presentamos a continuación las condiciones de optimalidad para el problema de minimización irrestricta
\begin{equation}
    \min f(x), \quad x \in \R^n.
    \label{v2:eq:1.4}
\end{equation}
Cabe notar que estos resultados son también verdaderos para un problema con restricciones
\[
    \min f(x) \quad \text{sujeto a} \quad x \in D,
\]
siempre que el punto de interés $\bar{x}$ esté en el interior del conjunto factible, i.e., $\bar{x} \in \interior D$.

\begin{theorem}[Condiciones de optimalidad en el caso irrestricto]\label{v2:thm:1.2.1}
    \begin{enumerate}[(a)]
        \item Supongamos que la función $f : \R^n \to \R$ sea diferenciable en el punto $\bar{x} \in \R^n$. Supongamos también que $\bar{x}$ sea un minimizador local del problema \eqref{v2:eq:1.4}. Entonces
        \begin{equation}
            f'(\bar{x}) = 0.
            \label{v2:eq:1.5}
        \end{equation}
        Si $f$ es dos veces diferenciable en $\bar{x}$, entonces además de \eqref{v2:eq:1.5} se tiene que la matriz Hessiana de $f$ en el punto $\bar{x}$ es semidefinida positiva, i.e.,
        \begin{equation}
            \langle f''(\bar{x})d, d \rangle \ge 0 \quad \forall d \in \R^n.
            \label{v2:eq:1.6}
        \end{equation}
        
        \item Supongamos que $f : \R^n \to \R$ sea dos veces diferenciable en el punto $\bar{x} \in \R^n$. Si $\bar{x}$ satisface \eqref{v2:eq:1.5} y si la matriz Hessiana de $f$ en $\bar{x}$ es definida positiva, i.e., si existe $\gamma > 0$ tal que
        \begin{equation}
            \langle f''(\bar{x})d, d \rangle \ge \gamma \|d\|^2 \quad \forall d \in \R^n,
            \label{v2:eq:1.7}
        \end{equation}
        entonces $\bar{x}$ es minimizador local estricto del problema \eqref{v2:eq:1.4}.
    \end{enumerate}
\end{theorem}

\begin{proof}
    Véanse los Teoremas 1.3.1--1.3.3 en \cite{izmailov2020otimizacaovolume1} (o equivalentemente, \cref{v1:thm:1.3.1,v1:thm:1.3.2,v1:thm:1.3.3}).
\end{proof}

La condición \eqref{v2:eq:1.5} se llama condición necesaria de primera orden para el problema \eqref{v2:eq:1.4}, y a los puntos que la satisfacen se les llama \textit{puntos estacionarios} de este problema. La combinación de \eqref{v2:eq:1.5} con \eqref{v2:eq:1.6} se llama condición necesaria de segunda orden, y la combinación de \eqref{v2:eq:1.5} con \eqref{v2:eq:1.7} es la condición suficiente de segunda orden para el problema \eqref{v2:eq:1.4}.

A continuación presentamos las condiciones de optimalidad para el caso de restricciones de igualdad:
\begin{equation}
    \min f(x) \quad \text{sujeto a} \quad x \in D = \{ x \in \R^n \mid h(x) = 0 \},
    \label{v2:eq:1.8}
\end{equation}
donde $f : \R^n \to \R$ y $h : \R^n \to \R^l$ son funciones dadas. La Lagrangiana del problema \eqref{v2:eq:1.8} es dada por
\[
    L : \R^n \times \R^l \to \R, \quad L(x, \lambda) = f(x) + \langle \lambda, h(x) \rangle = f(x) + \sum_{i=1}^l \lambda_i h_i(x).
\]
Observamos que
\[
    L'_x(x, \lambda) = f'(x) + (h'(x))^\top \lambda = f'(x) + \sum_{i=1}^l \lambda_i h'_i(x), \quad L'_\lambda(x, \lambda) = h(x).
\]
Las condiciones de optimalidad más útiles desde el punto de vista computacional son aquellas en forma primal-dual, en términos de la Lagrangiana del problema. Condiciones de este tipo necesitan de hipótesis de regularidad de las restricciones. Las condiciones de regularidad que serán utilizadas son:
\begin{equation}
    \{h'_i(\bar{x}), \; i = 1, \dots, l\} \quad \text{es un conjunto linealmente independiente,}
    \label{v2:eq:1.9}
\end{equation}
o bien
\begin{equation}
    h \text{ es una función afín, i.e., } h(x) = Ax - a, \text{ donde } A \in \R(l, n), \; a \in \R^l.
    \label{v2:eq:1.10}
\end{equation}
Bajo cualquiera de las hipótesis anteriores, tenemos el siguiente resultado.

\begin{theorem}[Condiciones de optimalidad en el caso de restricciones de igualdad]\label{v2:thm:1.2.2}
    \begin{enumerate}[(a)]
        \item Supongamos que $f : \R^n \to \R$ sea diferenciable en el punto $\bar{x} \in \R^n$ y que $h : \R^n \to \R^l$ sea diferenciable en una vecindad de este punto, con derivada continua en $\bar{x}$. Supongamos también que $\bar{x}$ sea un minimizador local del problema \eqref{v2:eq:1.8}. 
        Si vale una de las condiciones de regularidad de las restricciones (\eqref{v2:eq:1.9} o \eqref{v2:eq:1.10}), entonces existe $\bar{\lambda} \in \R^l$ tal que
        \begin{equation}
            L'_x(\bar{x}, \bar{\lambda}) = 0.
            \label{v2:eq:1.11}
        \end{equation}
        Bajo la condición \eqref{v2:eq:1.9}, el multiplicador $\bar{\lambda} \in \R^l$ que satisface \eqref{v2:eq:1.11} es único.
        Si $f$ y $h$ son dos veces diferenciables en el punto $\bar{x}$, entonces para cualquier $\bar{\lambda} \in \R^l$ que satisfaga \eqref{v2:eq:1.11} tenemos lo siguiente:
        \begin{equation}
            \langle L''_{xx}(\bar{x}, \bar{\lambda})d, d \rangle \ge 0 \quad \forall d \in \ker h'(\bar{x}).
            \label{v2:eq:1.12}
        \end{equation}
        
        \item Supongamos que $f : \R^n \to \R$ y $h : \R^n \to \R^l$ sean dos veces diferenciables en el punto $\bar{x} \in \R^n$.
        Si $\bar{x} \in D$ y existen $\bar{\lambda} \in \R^l$ satisfaciendo \eqref{v2:eq:1.11} y un número $\gamma > 0$ tales que
        \begin{equation}
            \langle L''_{xx}(\bar{x}, \bar{\lambda})d, d \rangle \ge \gamma \|d\|^2 \quad \forall d \in \ker h'(\bar{x}),
            \label{v2:eq:1.13}
        \end{equation}
        entonces $\bar{x}$ es un minimizador local estricto del problema \eqref{v2:eq:1.8}.
    \end{enumerate}
\end{theorem}

\begin{proof}
    Véanse los Teoremas 2.2.1, 2.3.1 y 2.3.2 en \cite{izmailov2020otimizacaovolume1} (o equivalentemente, \cref{v1:thm:2.2.1,v1:thm:2.3.1,v1:thm:2.3.2}).
\end{proof}

El sistema de ecuaciones
\[
    L'_x(x, \lambda) = 0, \quad h(x) = 0
\]
con respecto a $(x, \lambda) \in \R^n \times \R^l$, que caracteriza los puntos estacionarios del problema \eqref{v2:eq:1.8} y los multiplicadores de Lagrange asociados, se llama \textit{sistema de Lagrange}.

Finalmente, consideramos el caso más general, i.e., problemas con restricciones mixtas de igualdad y de desigualdad:
\begin{equation}
    \min f(x) \quad \text{sujeto a} \quad x \in D = \left\{ x \in \R^n \mathrel{\bigg|} \begin{array}{l} h(x) = 0, \\ g(x) \le 0 \end{array} \right\},
    \label{v2:eq:1.14}
\end{equation}
donde $f : \R^n \to \R$, $h : \R^n \to \R^l$ y $g : \R^n \to \R^m$ son funciones dadas. La Lagrangiana del problema \eqref{v2:eq:1.14} está dada por
\[
    L : \R^n \times \R^l \times \R^m \to \R, \quad L(x, \lambda, \mu) = f(x) + \langle \lambda, h(x) \rangle + \langle \mu, g(x) \rangle.
\]
En el caso de que haya restricciones de desigualdad, la siguiente noción es importante.

\begin{definition}\label{v2:def:1.2.1}
    Sea $\bar{x} \in D$, donde $D$ está definido en \eqref{v2:eq:1.14}. Decimos que la restricción de desigualdad que corresponde al índice $i \in \{1, \dots, m\}$ es \textbf{activa} en el punto $\bar{x}$ cuando $g_i(\bar{x}) = 0$, i.e., cuando la desigualdad es satisfecha como igualdad.
\end{definition}

El conjunto de los índices de las restricciones activas en el punto $\bar{x} \in D$ será denotado por
\[
    I(\bar{x}) = \{i = 1, \dots, m \mid g_i(\bar{x}) = 0\}.
\]
No incluimos en esta definición a las restricciones de igualdad porque todas ellas siempre son activas en todo punto factible.
Para representar las condiciones de optimalidad en forma primal-dual, precisaremos las hipótesis de regularidad de las restricciones.
De nuevo, una de las condiciones de este tipo es la linealidad de las restricciones:
\begin{equation}
    \begin{aligned}
    & h \text{ y } g \text{ son funciones afines, i.e.,} \\
    & h(x) = Ax - a, \; A \in \R(l, n), \; a \in \R^l, \\
    & g(x) = Bx - b, \; B \in \R(m, n), \; b \in \R^m.
    \end{aligned}
    \label{v2:eq:1.15}
\end{equation}
Otra condición de regularidad que es fundamental en la teoría de Optimización es la siguiente. Decimos que el punto $\bar{x} \in D$ satisface la \textbf{condición de regularidad de Mangasarian-Fromovitz} cuando
\begin{equation}
    \begin{gathered}
    \{h'_i(\bar{x}), \; i = 1, \dots, l\} \quad \text{es un conjunto linealmente independiente y} \\
    \exists d \in \ker h'(\bar{x}) \text{ tal que } \langle g'_i(\bar{x}), d \rangle < 0 \quad \forall i \in I(\bar{x}).
    \end{gathered}
    \label{v2:eq:1.16}
\end{equation}
Es fácil ver que cuando hay restricciones de desigualdad que son activas, esta condición es más débil (más general) que la de independencia lineal de los gradientes de las restricciones activas:
\begin{equation}
    \begin{gathered}
    \{h'_i(\bar{x}), \; i = 1, \dots, l\} \cup \{g'_i(\bar{x}), \; i \in I(\bar{x})\} \\
    \text{es un conjunto linealmente independiente.}
    \end{gathered}
    \label{v2:eq:1.17}
\end{equation}
Para el caso convexo (i.e., cuando $h$ es una función afín y $g$ es una función convexa), otra condición de calificación de las restricciones es la \textbf{condición de Slater}:
\begin{equation}
    \exists \hat{x} \in \R^n \text{ tal que } h(\hat{x}) = 0, \; g_i(\hat{x}) < 0, \; i = 1, \dots, m.
    \label{v2:eq:1.18}
\end{equation}

Para formular resultados de segunda orden precisaremos del llamado \textbf{cono crítico} (o cono de direcciones críticas) del problema \eqref{v2:eq:1.14} en el punto $x \in D$:
\begin{equation}
    K(x) = \left\{ d \in \R^n \mathrel{\bigg|} \begin{array}{l} \langle f'(x), d \rangle \le 0, \\ \langle g'_i(x), d \rangle \le 0 \; \forall i \in I(x), \\ \langle h'_i(x), d \rangle = 0, \; i = 1, \dots, l \end{array} \right\}.
    \label{v2:eq:1.19}
\end{equation}

\begin{theorem}[Condiciones de optimalidad en el caso de restricciones mixtas]\label{v2:thm:1.2.3}\ 
    \begin{enumerate}[(a)]
        \item Sean $f \colon \R^n \to \R$ y $g \colon \R^n \to \R^m$ diferenciables en el punto $\bar{x} \in \R^n$, y sea $h \colon \R^n \to \R^l$ diferenciable en una vecindad de $\bar{x}$, con derivada continua en este punto. Supongamos que $\bar{x}$ sea un minimizador local del problema \eqref{v2:eq:1.14}.
        Entonces, bajo cualquiera de las tres condiciones de regularidad de las restricciones (\eqref{v2:eq:1.15} o \eqref{v2:eq:1.16} o \eqref{v2:eq:1.18}), existen $\bar{\lambda} \in \R^l$ y $\bar{\mu} \in \R^m_+$ tales que
        \begin{align}
            L'_x(\bar{x}, \bar{\lambda}, \bar{\mu}) &= 0, \label{v2:eq:1.20} \\
            \bar{\mu}_i g_i(\bar{x}) &= 0, \quad i = 1, \dots, m. \label{v2:eq:1.21}
        \end{align}
        Bajo la condición \eqref{v2:eq:1.17}, el par $(\bar{\lambda}, \bar{\mu}) \in \R^l \times \R^m_+$ satisfaciendo \eqref{v2:eq:1.20} es único.
        Si $f$, $h$ y $g$ son dos veces diferenciables en $\bar{x}$ y satisface la condición de regularidad \eqref{v2:eq:1.17}, entonces para el único par de multiplicadores que satisface \eqref{v2:eq:1.20} y \eqref{v2:eq:1.21}, tenemos que
        \begin{equation}
            \langle L''_{xx}(\bar{x}, \bar{\lambda}, \bar{\mu})d, d \rangle \ge 0 \quad \forall d \in K(\bar{x}),
            \label{v2:eq:1.22}
        \end{equation}
        donde $K(\bar{x})$ es el cono crítico dado por \eqref{v2:eq:1.19}.
        
        \item Sean $f \colon \R^n \to \R$, $h \colon \R^n \to \R^l$ y $g \colon \R^n \to \R^m$ dos veces diferenciables en el punto $\bar{x} \in \R^n$.
        Si $\bar{x} \in D$ y existen $\bar{\lambda} \in \R^l$ y $\bar{\mu} \in \R^m_+$ satisfaciendo las condiciones \eqref{v2:eq:1.20} y \eqref{v2:eq:1.21}, tales que
        \begin{equation}
            \langle L''_{xx}(\bar{x}, \bar{\lambda}, \bar{\mu})d, d \rangle \ge \gamma \|d\|^2 \quad \forall d \in K(\bar{x}),
            \label{v2:eq:1.23}
        \end{equation}
        donde $\gamma > 0$ y $K(\bar{x})$ es el cono crítico dado por \eqref{v2:eq:1.19}, entonces $\bar{x}$ es un minimizador local estricto del problema \eqref{v2:eq:1.14}.
    \end{enumerate}
\end{theorem}

\begin{proof}
    Véanse los Teoremas 4.2.1, 4.3.1 y 4.3.2 en \cite{izmailov2020otimizacaovolume1} (o \cref{v1:thm:4.2.1,v1:thm:4.3.1,v1:thm:4.3.2}).
\end{proof}

La condición \eqref{v2:eq:1.21} se llama \textbf{condición de complementariedad}. Ella dice que
\[
    \bar{\mu}_i = 0 \quad \forall i \in \{1, \dots, m\} \setminus I(\bar{x}).
\]
En otras palabras, solamente los gradientes de las restricciones activas participan en la caracterización de optimalidad de un punto dado (lo que es, de cierto modo, intuitivamente obvio). Observamos aún que, como $g_i(\bar{x}) \le 0$ y $\bar{\mu}_i \ge 0$, la condición \eqref{v2:eq:1.21} es satisfecha si, y solamente si, $\langle \bar{\mu}, g(\bar{x}) \rangle = 0$. Además de eso, el conjunto de las relaciones $g_i(\bar{x}) \le 0$, $\bar{\mu}_i \ge 0$, $\bar{\mu}_i g_i(\bar{x}) = 0$, $i = 1, \dots, m$, puede ser escrito en la forma corta de un sistema de ecuaciones no lineales
\[
    \min\{-g_i(\bar{x}), \bar{\mu}_i\} = 0, \quad i = 1, \dots, m.
\]
Observamos que estas ecuaciones en general no son diferenciables.
El sistema de ecuaciones e inecuaciones
\[
    L'_x(x, \lambda, \mu) = 0, \quad h(x) = 0, \quad \mu \ge 0, \quad g(x) \le 0, \quad \langle \mu, g(x) \rangle = 0
\]
en las variables $(x, \lambda, \mu) \in \R^n \times \R^l \times \R^m$, caracteriza los puntos estacionarios del problema \eqref{v2:eq:1.14} y los multiplicadores de Lagrange asociados. Este sistema se llama \textbf{sistema de Karush-Kuhn-Tucker (KKT)}.

Cuando $\bar{x}$ es un punto estacionario del problema \eqref{v2:eq:1.14}, para todo par $(\bar{\lambda}, \bar{\mu}) \in \R^l \times \R^m_+$ de multiplicadores de Lagrange asociados a $\bar{x}$, el cono crítico puede ser escrito de manera equivalente como
\begin{equation}
    K(\bar{x}) = \left\{ d \in \R^n \mathrel{\bigg|} \begin{array}{l} \langle h'_i(\bar{x}), d \rangle = 0, \; i = 1, \dots, l, \\ \langle g'_i(\bar{x}), d \rangle \le 0 \; \forall i \in I(\bar{x}), \\ \bar{\mu}_i \langle g'_i(\bar{x}), d \rangle = 0 \; \forall i \in I(\bar{x}) \end{array} \right\}.
    \label{v2:eq:1.24}
\end{equation}
Cuando
\begin{equation}
    \bar{\mu}_i > 0 \quad \forall i \in I(\bar{x}),
    \label{v2:eq:1.25}
\end{equation}
decimos que el punto estacionario $\bar{x}$ satisface la \textbf{condición de complementariedad estricta} con multiplicadores de Lagrange $\bar{\lambda}$ y $\bar{\mu}$. Observamos que, bajo esta condición, \eqref{v2:eq:1.24} se reduce a
\[
    K(\bar{x}) = \{ d \in \ker h'(\bar{x}) \mid \langle g'_i(\bar{x}), d \rangle = 0 \; \forall i \in I(\bar{x}) \}.
\]

Concluimos con la siguiente caracterización de compacidad del conjunto de multiplicadores de Lagrange.

\begin{proposition}\label{v2:prop:1.2.1}
    Sea $\bar{x}$ un punto estacionario del problema \eqref{v2:eq:1.14} (en particular, el conjunto de multiplicadores de Lagrange asociados a $\bar{x}$ no es vacío).
    El conjunto de multiplicadores es compacto si, y solamente si, la condición de regularidad de Mangasarian-Fromovitz \eqref{v2:eq:1.16} es satisfecha.
\end{proposition}

\begin{proof}
    Véase el Teorema 4.2.2 en \cite{izmailov2020otimizacaovolume1} (o \cref{v1:thm:4.2.2}).
\end{proof}
