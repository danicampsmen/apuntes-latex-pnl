\subsection{Penalización interna. Métodos de barreras}\label{v2:sec:4.3.3}
% =========================================================

Los métodos de penalización interna lidian con problemas de la forma
\begin{equation}
    \min f(x) \quad \text{sujeto a } x \in D = \{x \in \Rn \mid g(x) \le 0\},
    \label{v2:eq:4.83}
\end{equation}
donde $f : \Rn \to \R$, $g : \Rn \to \R^m$, y el ``interior'' del conjunto factible no es vacío:
\begin{equation}
    D^0 = \{x \in \Rn \mid g(x) < 0\} \neq \emptyset.
    \label{v2:eq:4.84}
\end{equation}
Supongamos, aún, que
\begin{equation}
    \inf_{x \in D^0} f(x) = \inf_{x \in D} f(x) = \bar{v} > -\infty,
    \label{v2:eq:4.85}
\end{equation}
i.e., el ínfimo de $f$ en el conjunto $D^0$ es igual al valor óptimo del problema \eqref{v2:eq:4.83}. Esta hipótesis tiene por objetivo eliminar ciertos casos patológicos, como aquel en la Figura \ref{v2:fig:4.3.5}, donde, evidentemente, la solución $\bar{x}$ no puede ser aproximada por puntos del interior del conjunto factible.

    \centering
    % \includegraphics[width=0.6\textwidth]{figures/fig_4_3_5.png}
    \caption{La solución $\bar{x}$ no puede ser aproximada por puntos del interior del conjunto factible $D$.}
    \label{v2:fig:4.3.5}
\end{figure}

La idea de métodos de penalización interna es la misma que la de penalización externa, en el sentido de que el problema original es aproximado por una secuencia de problemas que pueden ser considerados irrestrictos. Solo que este objetivo es llevado a cabo de otra manera. En particular, a la función objetivo se le añade un término penalizador que tiende a más infinito para cualquier secuencia que se aproxime a la frontera $D \setminus D^0$ del conjunto factible.

\begin{definition}\label{v2:def:4.3.1}
    Una función $\psi : D^0 \to \R_+$ se llama \textit{barrera} para el conjunto $D$ definido en \eqref{v2:eq:4.83}, si ella es continua en $D^0$, y para toda secuencia $\{x^k\} \in D^0$ tal que $g_i(x^k) \to 0$ para algún $i \in \{1, \dots, m\}$, se tiene que $\psi(x^k) \to +\infty \ (k \to \infty)$.
\end{definition}

Las elecciones más comunes son la \textit{barrera logarítmica}
\begin{equation}
    \psi(x) = -\sum_{i=1}^m \log(-g_i(x)), \quad x \in D^0,
    \label{v2:eq:4.86}
\end{equation}
y la \textit{barrera inversa}
\begin{equation}
    \psi(x) = -\sum_{i=1}^m 1/g_i(x), \quad x \in D^0.
    \label{v2:eq:4.87}
\end{equation}

Consideremos una familia de funciones
\begin{equation}
    \varphi(\cdot; t) : D^0 \to \R, \quad \varphi(x; t) = f(x) + t\psi(x),
    \label{v2:eq:4.88}
\end{equation}
donde $t > 0$ es el parámetro penalizador, y los problemas
\begin{equation}
    \min \varphi(x; t) \quad \text{sujeto a } x \in D^0.
    \label{v2:eq:4.89}
\end{equation}
Podemos pensar en el problema \eqref{v2:eq:4.89} como de optimización irrestricta en el sentido siguiente. Es bien claro que si simplemente ignoramos por completo las restricciones del problema \eqref{v2:eq:4.89} (que son desigualdades estrictas) y resolvemos el problema
\[
    \min \varphi(x; t), \quad x \in \Rn,
\]
aplicando cualquier método iterativo razonable a partir de un punto inicial en $D^0$, toda la secuencia generada por este método (así como sus puntos de acumulación) va a quedar dentro de $D^0$, ya que los puntos cerca de la frontera $D \setminus D^0$ tienen valores de $\varphi(\cdot; t)$ explosivos y, por lo tanto, no son nada atractivos para algoritmos de minimización.

\vspace{0.3cm}
\noindent\textbf{Algoritmo 4.3.2. (Método de barreras)}\\
Elegir una función barrera $\psi : D^0 \to \R$ para el conjunto $D$ definido en \eqref{v2:eq:4.83}. Elegir $t_0 > 0$ y tomar $k := 0$.
\begin{enumerate}
    \item Calcular $x^k \in D^0$ como una solución del problema \eqref{v2:eq:4.89}, donde la función objetivo es dada por la fórmula \eqref{v2:eq:4.88} con $t = t_k$.
    \item Elegir $t_{k+1} < t_k$ y tomar $k := k + 1$. Retornar al Paso 1.
\end{enumerate}
\vspace{0.3cm}

Observemos que como las soluciones del problema original típicamente están en la frontera $D \setminus D^0$, queda claro que para la convergencia de métodos de este tipo es necesario que los valores del parámetro $t$ tiendan a cero (para amenizar la influencia explosiva de la barrera $\psi$ y permitir la aproximación a soluciones en la frontera del conjunto factible).

\begin{example}\label{v2:exa:4.3.4}
    Consideremos el problema
    \[
        \min x \quad \text{sujeto a } x \ge 0.
    \]
    Este es un problema de programación convexa y su única solución es $\bar{x} = 0$.
    Eligiendo la barrera inversa \eqref{v2:eq:4.87}, tenemos
    \[
        \varphi(x; t) = x + t/x, \quad t > 0.
    \]
    Los puntos estacionarios de esta función en el conjunto abierto
    \[
        D^0 = \{x \in \R \mid x > 0\} \neq \emptyset
    \]
    están caracterizados por la ecuación
    \[
        0 = \varphi'(x; c) = 1 - t/x^2,
    \]
    de donde obtenemos
    \[
        x(t) = \sqrt{t} \in D^0.
    \]
    En particular, $x(t) \to 0 = \bar{x}$ cuando $t \to 0$.
\end{example}

    \caption{Para las soluciones de los problemas penalizados, se tiene que $x(t) \in D^0$ para todo $t > 0$ y $x(t) \to \bar{x}$ cuando $t \to 0+$.}

\end{figure}

\begin{example}\label{v2:exa:4.3.5}
    Consideremos el problema
    \[
        \min x_1 + x_2 \quad \text{sujeto a } -x_1^2 + x_2 \ge 0, \ x_1 \ge 0.
    \]
    Este es un problema de programación convexa y su única solución es $\bar{x} = 0$.
    Eligiendo la barrera logarítmica \eqref{v2:eq:4.86}, tenemos
    \[
        \varphi(x; t) = x_1 + x_2 - t \log(-x_1^2 + x_2) - t \log(x_1).
    \]
    Los puntos estacionarios del problema de minimizar esta función en el conjunto abierto
    \[
        D^0 = \{x \in \R^2 \mid -x_1^2 + x_2 > 0, \ x_1 > 0\} \neq \emptyset
    \]
    están caracterizados por el sistema de ecuaciones
    \[
        0 = \varphi'(x; t) = \begin{pmatrix} 1 + 2tx_1/(-x_1^2 + x_2) - t/x_1 \\ 1 - t/(-x_1^2 + x_2) \end{pmatrix}.
    \]
    De donde obtenemos
    \[
        x_1(t) = (-1 + \sqrt{1 + 8t})/4, \quad x_2(t) = t + ((-1 + \sqrt{1 + 8t})/4)^2.
    \]
    En particular, $x(t) \to 0 = \bar{x}$ cuando $t \to 0$.
    La trayectoria de soluciones de los problemas penalizados es ilustrada en la Figura \ref{v2:fig:4.3.6}.
\end{example}

Ahora estudiaremos propiedades de convergencia del Algoritmo 4.3.2. Como en el caso de la penalización externa, los resultados de convergencia para métodos de barreras se refieren a la secuencia generada por minimizadores globales de problemas penalizados.

\begin{proposition}\label{v2:prop:4.3.3}
    Supongamos que el conjunto $D$ definido en \eqref{v2:eq:4.83} satisface \eqref{v2:eq:4.84} y \eqref{v2:eq:4.85}. Sea $\{x^k\}$ una secuencia generada por el Algoritmo 4.3.2, donde $x^k$ es una solución global del problema \eqref{v2:eq:4.89} con $t = t_k$.
    Entonces, para todo $k$, se tiene que
    \begin{align}
        \varphi(x^{k+1}; t_{k+1}) &\le \varphi(x^k; t_k), \label{v2:eq:4.90} \\
        \psi(x^{k+1}) &\ge \psi(x^k), \label{v2:eq:4.91} \\
        f(x^{k+1}) &\le f(x^k). \label{v2:eq:4.92}
    \end{align}
\end{proposition}

\begin{proof}
    Por la otimalidad de $x^{k+1}$ en \eqref{v2:eq:4.89}, $t = t_{k+1}$, tenemos que
    \begin{align*}
        f(x^{k+1}) + t_{k+1}\psi(x^{k+1}) &= \varphi(x^{k+1}; t_{k+1}) \\
        &\le \varphi(x^k; t_{k+1}) \\
        &= f(x^k) + t_{k+1}\psi(x^k).
    \end{align*}
    Sumando esta última desigualdad a la desigualdad \eqref{v2:eq:4.93} (que obtuve del paso análogo en el texto original, vea a seguir), obtenemos que
    \[
        (t_k - t_{k+1})\psi(x^k) \le (t_k - t_{k+1})\psi(x^{k+1}).
    \]
    De donde, como $t_{k+1} < t_k$, sigue \eqref{v2:eq:4.91}.
    Ahora, utilizando de nuevo la relación que probamos, obtenemos
    \[
        f(x^{k+1}) - f(x^k) \le t_{k+1}(\psi(x^k) - \psi(x^{k+1})) \le 0,
    \]
    donde la segunda desigualdad sigue de $t_{k+1} > 0$ y de \eqref{v2:eq:4.91}.
    Finalmente, obtenemos el siguiente resultado.
\end{proof}

\begin{theorem}[Convergencia de los métodos de penalización interna]\label{v2:thm:4.3.7}
    Sean $f : \Rn \to \R$ y $g : \Rn \to \R^m$ funciones continuas en $\Rn$. Supongamos que el conjunto $D$ definido en \eqref{v2:eq:4.83} satisface \eqref{v2:eq:4.84} y \eqref{v2:eq:4.85}.
    Entonces todo punto de acumulación de la secuencia $\{x^k\}$ generada por el Algoritmo 4.3.2, donde $x^k$ es una solución global del problema \eqref{v2:eq:4.89} con $t = t_k$ y $t_k \to 0 \ (k \to \infty)$, es una solución global del problema \eqref{v2:eq:4.83}.
\end{theorem}

\begin{proof}
    Sea $\bar{x}$ un punto de acumulación de la secuencia $\{x^k\}$. Sea $\{x^{k_j}\} \to \bar{x} \ (j \to \infty)$. Como $\{x^k\} \subset D^0$, i.e., tenemos que $g(x^k) < 0$ para todo $k$, y $g$ es continua, concluimos que $g(\bar{x}) \le 0$. O sea, $\bar{x} \in D$.
    Si $f(\bar{x}) = \bar{v}$, no hay nada más que probar. Sea $f(\bar{x}) > \bar{v}$. Definimos
    \begin{equation}
        \delta = (f(\bar{x}) - \bar{v})/2 > 0.
        \label{v2:eq:4.94}
    \end{equation}
    Por la relación \eqref{v2:eq:4.85},
    \begin{equation}
        \exists \tilde{x} \in D^0 \text{ tal que } f(\tilde{x}) < \bar{v} + \delta.
        \label{v2:eq:4.95}
    \end{equation}
    La secuencia $\{f(x^k)\}$ es no-creciente (vea \eqref{v2:eq:4.92}). Por lo tanto, $\{f(x^{k_j})\}$ también es no-creciente y tenemos que
    \[
        f(x^{k_j}) \ge \lim_{j \to \infty} f(x^{k_j}) = f(\bar{x}) = \bar{v} + 2\delta,
    \]
    donde llevamos en cuenta la continuidad de $f$ y \eqref{v2:eq:4.94}.
    Ahora, utilizando \eqref{v2:eq:4.95}, obtenemos que
    \begin{equation}
        f(x^{k_j}) - f(\tilde{x}) \ge \bar{v} + 2\delta - f(\tilde{x}) > \delta.
        \label{v2:eq:4.96}
    \end{equation}
    Tenemos entonces que, para $j$ suficientemente grande,
    \begin{align*}
        0 &\ge \varphi(x^{k_j}; t_{k_j}) - \varphi(\tilde{x}; t_{k_j}) \\
        &= f(x^{k_j}) + t_{k_j}\psi(x^{k_j}) - f(\tilde{x}) - t_{k_j}\psi(\tilde{x}) \\
        &> \delta + t_{k_j}(\psi(x^{k_j}) - \psi(\tilde{x})) \\
        &\ge \delta + t_{k_j}(\psi(x^{k_0}) - \psi(\tilde{x})) \\
        &\ge \delta/2 > 0,
    \end{align*}
    donde la primera desigualdad sigue del hecho de que $x^{k_j}$ es una solución global del problema \eqref{v2:eq:4.89} con $t = t_{k_j}$, la segunda desigualdad sigue de \eqref{v2:eq:4.96}, la tercera de la desigualdad \eqref{v2:eq:4.91}, y la última del hecho de que $t_k \to 0 \ (k \to \infty)$.
    La contradicción encima implica que $f(\bar{x}) = \bar{v}$, lo que concluye la demostración.
\end{proof}
