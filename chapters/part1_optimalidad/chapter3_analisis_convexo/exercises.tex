\addsec{Ejercicios del Capítulo}

\addsubsec{Parte I: Teoremas de Alternativa}

\begin{exercise}[Teorema de Tucker]\label{v1:ejer:3.1}
 Pruebe el Teorema de la Alternativa de Tucker: Para cualesquiera matrices $A_i$, $i=0,1,2$, con $A_0 \neq 0$, uno y solo uno de los sistemas posee solución:
 \[
 A_0 x \ge 0 \ (\text{con } A_0 x \neq 0), \quad A_1 x = 0, \quad A_2 x \ge 0, \quad x \in \Rn;
 \]
 o
 \[
 A_0^\top y^0 + A_1^\top y^1 + A_2^\top y^2 = 0, \quad y^0 > 0, \ y^2 \ge 0.
 \]
\end{exercise}

\begin{exercise}[Teoremas de Gale I y II]\label{v1:ejer:3.2}
 Demuestre los Teoremas de Alternativa de Gale:
 \begin{enuthm}
 \item $Ax = a \ (x \ge 0)$ vs $A^\top y \ge 0 \ (\interno{a}{y} < 0)$.
 \item $Ax \le a \ (x \ge 0)$ vs $A^\top y \ge 0 \ (\interno{a}{y} < 0, \ y \ge 0)$.
 \end{enuthm}
\end{exercise}

\begin{exercise}[Teoremas de Gordan y Stiemke]\label{v1:ejer:3.3}
 Demuestre los Teoremas de Alternativa:
 \begin{enuthm}
 \item \textbf{Gordan:} $Ax < 0$ vs $A^\top y = 0 \ (y \ge 0, y \neq 0)$.
 \item \textbf{Stiemke:} $Ax \le 0 \ (Ax \neq 0)$ vs $A^\top y = 0 \ (y > 0)$.
 \end{enuthm}
\end{exercise}

\addsubsec{Parte II: Análisis Convexo}

\begin{exercise}[Desigualdad de la media y Cobb-Douglas]\label{v1:ejer:3.4}
 Sean $\alpha_i > 0, \ i=1,\dots,p$. Probar utilizando propiedades de funciones convexas:
 \[
 \frac{1}{p}\sum_{i=1}^p \alpha_i \ge \left(\prod_{i=1}^p \alpha_i\right)^{1/p}.
 \]
\end{exercise}

\begin{exercise}[Módulo de convexidad fuerte]\label{v1:ejer:3.5}
 Probar que la función $f(x) = \norm{x}^2$ es fuertemente convexa con módulo $\gamma = 1$.
\end{exercise}

\begin{exercise}[Linealidad afín]\label{v1:ejer:3.6}
 Sea $f$ una función convexa y al mismo tiempo cóncava en $\Rn$. Mostrar que esto implica que $f$ es una función afín (es decir, $f(x) = \interno{a}{x} + c$).
\end{exercise}

\begin{exercise}[Condiciones paramétricas de convexidad]\label{v1:ejer:3.7}
 ¿Para qué valores del parámetro $\sigma \in \R$ la función $f(x) = (\sigma + 6)x_1^2 + 2\sigma x_1 x_2 + x_2^2$ es convexa en $\R^2$?
\end{exercise}

\begin{exercise}[Funciones generadoras convexas]\label{v1:ejer:3.8}
 Probar que las siguientes funciones son convexas en $\Rn$:
 \begin{enuthm}
 \item $f(x) = \ln\left(\sum_{i=1}^n \E^{x_i}\right)$.
 \item $f(x) = \E^{\interno{Qx}{x}}$, donde $Q$ es una matriz simétrica y semidefinida positiva.
 \end{enuthm}
\end{exercise}

\begin{exercise}[Minimización analítica en cajas]\label{v1:ejer:3.9}
 Resolver de forma analítica el problema $\min f(x)$ sujeto a $x \in D$:
 \begin{enuthm}
 \item $f(x) = 2x_1^2 + 4x_2^2 - 5x_1 x_2, \quad D = \{x \in \R^2 \mid 0 \le x_1 \le 1, \ 0 \le x_2 \le 1\}$.
 \item $f(x) = 4x_1^2 - x_1 x_2 + 2x_2^2, \quad D = \{x \in \R^2 \mid 4 \le x_1 \le 8, \ -1 \le x_2 \le 2\}$.
 \end{enuthm}
\end{exercise}

\begin{exercise}[Operaciones que preservan convexidad]\label{v1:ejer:3.10}
 Probar que la función $f(x) = \max\{ \E^x, x^2 \} + 1/x$ es convexa en el dominio $\R_+ \setminus \{0\}$.
\end{exercise}
